from __future__ import annotations

from collections import abc
from typing import Iterable, Union

import deepgenomics.genomekit.v1 as messages
from typing_extensions import Literal

from . import genome, genome_annotation, interval, variant

GenomeKitObject = Union[
    "interval.Interval",
    "variant.Variant",
    "genome.Genome",
    "genome_annotation.Gene",
    "genome_annotation.Transcript",
    "genome_annotation.Exon",
    "genome_annotation.Intron",
    "genome_annotation.Cds",
    Iterable["interval.Interval"],
]

_TYPE_FIELD_TABLE = {
    genome_annotation.Cds: "cds",
    genome_annotation.Exon: "exon",
    genome_annotation.Gene: "gene",
    genome.Genome: "genome",
    interval.Interval: "interval",
    genome_annotation.Intron: "intron",
    genome_annotation.Transcript: "transcript",
    variant.Variant: "variant",
}
_FIELD_TYPE_TABLE = {v: k for k, v in _TYPE_FIELD_TABLE.items()}



[docs]def marshal(
    obj: GenomeKitObject, field_fill: Literal["minimal", "full"] = "minimal"
) -> bytes:
    """Returns a state string suitable for fast transmission or serialization.

    Note that objects can be directly pickled, without calling :func:`~genome_kit.marshal`.
    This method is mainly provided to facilitate use with the `marshal` package.
    The result is about 20x faster and 30% smaller than `cPickle`::

        >>> objs       = [ ... list of objs ... ]
        >>> pickled    = cPickle.dumps(objs)               # Simple, general
        >>> marshalled = marshal.dumps([gk.marshal(obj)
                                        for obj in objs])  # Faster, smaller

    Parameters
    ----------
    obj
        GenomeKit object to marshal

    field_fill
        Determines how much information to store. 'full' will fill every
        field (the default is 'minimal', which only fills enough to use
        :func:`~genome_kit.unmarshal`)

    Returns
    -------
    bytes
        A state string encoding this object.
    """
    return as_proto(obj, field_fill).SerializeToString()




[docs]def unmarshal(state: bytes) -> GenomeKitObject:
    """Creates an instance by unmarshalling a state string.

    Parameters
    ----------
    state
        The state string, previously returned by :func:`~genome_kit.marshal`.

    Returns
    -------
    GenomeKitObject
        A new object instance corresponding to the state string.
    """
    message = messages.TaggedUnion()
    message.ParseFromString(state)
    return from_proto(message)




[docs]def as_proto(
    obj: GenomeKitObject, field_fill: Literal["minimal", "full"] = "minimal"
) -> messages.TaggedUnion:
    """Returns a protobuf message encoding this instance.

    Parameters
    ----------
    obj
        GenomeKit object to encode

    field_fill
        Determines how much information to store. 'full' will fill every
        field (the default is 'minimal', which only fills enough to use
        :func:`~genome_kit.from_proto`)

    Returns
    -------
    messages.TaggedUnion
        A protobuf message encoding this object.
    """
    message = messages.TaggedUnion()
    if isinstance(obj, abc.Iterable):
        for x in obj:
            x.as_proto(message.disjoint_interval.interval.add(), field_fill)
    else:
        obj.as_proto(getattr(message, _TYPE_FIELD_TABLE[type(obj)]), field_fill)
    return message




[docs]def from_proto(message: messages.TaggedUnion) -> GenomeKitObject:
    """Creates an instance by decoding a protobuf message.

    Parameters
    ----------
    message
        The protobuf, previously returned by :func:`~genome_kit.as_proto`.

    Returns
    -------
    GenomeKitObject
        A new object instance corresponding to the protobuf message.
    """
    field = message.WhichOneof("type")
    message = getattr(message, field)
    if field == "disjoint_interval":
        return [interval.Interval.from_proto(x) for x in message.interval]
    return _FIELD_TYPE_TABLE[field].from_proto(message)
